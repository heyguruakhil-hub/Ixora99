🔗 Google Sheets sync

Push live employee data to a Google Sheet automatically

This uses a Google Apps Script Web App that you deploy once — it acts as the bridge between Workfloor and your Google Sheet. No API keys or OAuth needed.

Step 1 — Create a Google Sheet

Open Google Sheets, create a new spreadsheet and name it Workfloor Tracker.

Step 2 — Add the Apps Script

In your sheet, go to Extensions → Apps Script. Select ALL existing code (Ctrl+A) and DELETE it completely, then paste this fresh:

⚠️ Important: The script must only contain doGet — NOT doPost. If you see doPost anywhere, delete it. That is what causes the "could not save" error. 
const SHEET_NAME = "Daily Records";
const ATTENDANCE_SHEET = "Attendance";
const EMPLOYEES_SHEET = "Employees";

// IMPORTANT: This uses doGet (not doPost) to avoid CORS issues
function doGet(e) {
  try {
    const action = e.parameter.action;

    if (action === "ping") {
      return jsonResponse({ok: true, status: "Workfloor bridge ready"});
    }

    if (action === "syncAll") {
      const payload = JSON.parse(decodeURIComponent(e.parameter.data));
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      syncDailyRecords(ss, payload.days, payload.employees);
      syncAttendance(ss, payload.attendance, payload.employees);
      syncEmployees(ss, payload.employees);
      return jsonResponse({ok: true, message: "Synced " + new Date().toLocaleString()});
    }

    return jsonResponse({ok: false, error: "Unknown action: " + action});
  } catch(err) {
    return jsonResponse({ok: false, error: err.toString()});
  }
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function syncDailyRecords(ss, days, employees) {
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) sheet = ss.insertSheet(SHEET_NAME);
  sheet.clearContents();
  const headers = ["Date","Employee","Login","Logout","Total Count","Task Type","Parent Task","Task","Task Start","Task End","Duration (min)","Duration","Count","Done"];
  sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight("bold");
  const rows = [];
  Object.keys(days).sort().forEach(date => {
    employees.forEach(emp => {
      const day = days[date] && days[date][emp.id];
      if (!day) return;
      const total = day.tasks.filter(t => !t.parentId).reduce((s,t)=>s+(parseInt(t.count)||0),0);
      if (day.tasks.length === 0) {
        rows.push([date,emp.name,day.login||"",day.logout||"",total,"","","","","","","","",""]);
      } else {
        day.tasks.forEach(t => {
          const durMin = t.duration != null ? t.duration : "";
          const durHM = t.duration != null ? Math.floor(t.duration/60)+"h "+(t.duration%60)+"m" : "";
          const isSub = !!t.parentId;
          const parentName = isSub ? ((day.tasks.find(p => p.id === t.parentId) || {}).name || "") : "";
          const taskLabel = isSub ? ("\u21b3 " + t.name) : t.name;
          rows.push([date,emp.name,day.login||"",day.logout||"",total,isSub?"Subtask":"Task",parentName,taskLabel,t.startTime||"",t.endTime||"",durMin,durHM,t.count||0,t.done?"Yes":"No"]);
        });
      }
    });
  });
  if (rows.length) sheet.getRange(2,1,rows.length,headers.length).setValues(rows);
  autoResize(sheet, headers.length);
}

function syncAttendance(ss, attendance, employees) {
  let sheet = ss.getSheetByName(ATTENDANCE_SHEET);
  if (!sheet) sheet = ss.insertSheet(ATTENDANCE_SHEET);
  sheet.clearContents();
  const headers = ["Date","Employee","Status","Note"];
  sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight("bold");
  const rows = [];
  employees.forEach(emp => {
    const empAttendance = (attendance && attendance[emp.id]) || {};
    Object.keys(empAttendance).sort().forEach(date => {
      const rec = empAttendance[date];
      if (!rec || !rec.status) return;
      rows.push([date, emp.name, rec.status||"", rec.note||""]);
    });
  });
  rows.sort((a,b)=> a[0]b[0]?1:0);
  if (rows.length) sheet.getRange(2,1,rows.length,headers.length).setValues(rows);
  autoResize(sheet, headers.length);
}

function syncEmployees(ss, employees) {
  let sheet = ss.getSheetByName(EMPLOYEES_SHEET);
  if (!sheet) sheet = ss.insertSheet(EMPLOYEES_SHEET);
  sheet.clearContents();
  const headers = ["ID","Name","Username","Email"];
  sheet.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight("bold");
  const rows = employees.map(e => [e.id, e.name, e.username, e.email||""]);
  if (rows.length) sheet.getRange(2,1,rows.length,headers.length).setValues(rows);
  autoResize(sheet, headers.length);
}

function autoResize(sheet, numCols) {
  for (let i=1; i<=numCols; i++) sheet.autoResizeColumn(i);
}

Step 3 — Deploy as Web App

⚠️ If you deployed before and got an error — do a NEW deployment, do not edit the old one.
  1. Click Deploy → New deployment (top-right button)
  2. Click the gear icon ⚙️ and select type: Web app
  3. Set Execute as: Me
  4. Set Who has access: Anyone (not "Anyone with Google account")
  5. Click Deploy → Authorize → copy the Web app URL
